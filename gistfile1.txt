class Player:
    def __init__(self, name, lives):
        self.name = name
        self._lives = lives

    def health(self):
        self.current = self._lives
        print(f'Current HP of {self.name}: {self.current}')

    def hit(self):
        self._lives -= 1
        if self._lives == 0:
            print('Game Over')
